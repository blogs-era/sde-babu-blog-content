class Solution {
    public TreeNode getRightMostNode(TreeNode root){
        while(root.right!=null){
            root = root.right;
        }
        return root;
    }
    public TreeNode deleteNodeUtil(TreeNode root){
        if(root.left==null)return root.right;
        if(root.right==null)return root.left;
        TreeNode leftChild = root.left;
        TreeNode rightMostNode = getRightMostNode(leftChild);
        rightMostNode.right = root.right;
        return root.left;
    }
    public TreeNode deleteNode(TreeNode root, int key) {
        // just note that in this algo we are just doing some important steps:
        // (1) try to find the key in tree which we have to delete
        // (2) reach to the rightmost part on left side: rightMostPart
        // (3) rightMostPart.right = node_to_be_deleted.right
        // (4) and just connect the parent to node_to_be_deleted.left
        // (5) connection of parent can be either parent.left = node_to_be_deleted.left or parent.right = node_to_be_deleted.left depend on the key on which side of parent
        if(root==null)return null;
        if(root.val==key){
            return deleteNodeUtil(root);
        }
        TreeNode dummy = root;
        while(root!=null){
            // move to the left node
            if(root.val > key){
                if(root.left!=null && root.left.val==key){
                    // left node is the key
                    root.left = deleteNodeUtil(root.left);
                    break;
                }else{
                    root = root.left;
                }
            }
            // move to the right node
            else{
                if(root.right!=null && root.right.val==key){
                    // right node is the key
                    root.right = deleteNodeUtil(root.right);
                    break;
                }else{
                    root = root.right;
                }
            }
        }
        return dummy;
    }
}
