class BSTIterator {
    Stack<TreeNode> stack;// stack to store the node 
    public BSTIterator(TreeNode root) {
        stack = new Stack<>();
        // store and move to left node until we end up with next node
        while(root!=null){
            stack.push(root);
            root = root.left;
        }
    }
    
    public int next() {
        // it is mentioned that next function only called when there exist a next element  
        TreeNode top = stack.peek();// store the top element because this is the next element 
        TreeNode root = top.right;// go to right node 
        stack.pop();
        // store and move to left node until we end up with next node
        while(root!=null){
            stack.push(root);
            root = root.left;
        }
        return top.val;
    }
    
    public boolean hasNext() {
        // if there is element in stack then true
        return !stack.isEmpty();
    }
}
