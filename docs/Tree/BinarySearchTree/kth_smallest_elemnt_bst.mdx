```js
    // count and ans
    int count = 1,ans = -1;
    // is_added to avoid any other ans you can test this by removing is_added==false condition
    boolean is_added = false;
    public void dfs(TreeNode root,int k){
        //base case
        if(root==null)return;
        // in order
        dfs(root.left,k);
        // check 
        if(count==k && is_added==false){
            ans = root.val;
            is_added = true;
            return;
        }
        count++;        
        dfs(root.right,k);
    }
    public int kthSmallest(TreeNode root, int k) {
        dfs(root,k);
        return ans;
    }

```
