/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<TreeNode> generate(int start,int end){
        List<TreeNode> res = new ArrayList<>(); 
        if(start > end){
            res.add(null);
            return res;           
        }
        for(int i=start;i<=end;i++){
            List<TreeNode> left_roots = generate(start,i-1);
            List<TreeNode> right_roots = generate(i+1,end);
            for(TreeNode left_root: left_roots){
                for(TreeNode right_root: right_roots){
                    TreeNode root = new TreeNode(i);// root node
                    root.left = left_root;
                    root.right = right_root;
                    res.add(root);
                }
            }
        }
        return res;
    } 
    public List<TreeNode> generateTrees(int n) {
        List<TreeNode> result = generate(1,n);
        return result;
    }
}
