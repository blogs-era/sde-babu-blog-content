/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    TreeNode lca(TreeNode root,int a,int b){
        if(root==null)return null;
        if(root.val==a || root.val==b)return root;
        
        TreeNode left = lca(root.left,a,b);
        TreeNode right = lca(root.right,a,b);
        
        if(left!=null && right!=null)return root;
        if(left!=null)return left;
        if(right!=null)return right;
        
        return null;   
    }
    public String pathFromStart(TreeNode root,int startValue,StringBuilder path){
        if(root==null)return "";
        if(root.val==startValue)return path.toString();
        String left = pathFromStart(root.left,startValue,path.append("U")).toString();
        path.deleteCharAt(path.length()-1);
        String right = pathFromStart(root.right,startValue,path.append("U")).toString();
        path.deleteCharAt(path.length()-1);
        return left + right;
    }  
    public String pathFromDest(TreeNode root,int destValue,StringBuilder path){
        if(root==null)return "";
        if(root.val==destValue)return path.toString();
        String left = pathFromDest(root.left,destValue,path.append("L")).toString();
        path.deleteCharAt(path.length()-1);
        String right = pathFromDest(root.right,destValue,path.append("R")).toString();
        path.deleteCharAt(path.length()-1);
        return left + right;
    }  
    public String getDirections(TreeNode root, int startValue, int destValue) {
        TreeNode lowest_common_ancestor = lca(root,startValue,destValue);   
        return pathFromStart(lowest_common_ancestor,startValue,new StringBuilder("")) + pathFromDest(lowest_common_ancestor,destValue,new StringBuilder(""));
    }
}
