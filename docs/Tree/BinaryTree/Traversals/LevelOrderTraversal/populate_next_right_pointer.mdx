public Node connect(Node root) {
    if(root==null)return null;
    Queue<Node> q = new LinkedList<>();// init the queue
    q.add(root);// add the root node in the queue
    
    while(!q.isEmpty()){
        int size = q.size();// get the size of the level 
        Node prev = null;// track prev for next right ptr 
        for(int i=0;i<size;i++){
            Node top = q.poll();
            if(prev!=null)prev.next = top;//connect ptr
            prev = top;// store the prev
            // if left is not null put in the queue
            if(top.left!=null)q.add(top.left);
            // if right is not null put in the queue
            if(top.right!=null)q.add(top.right);
        }
    }
    return root;
}
