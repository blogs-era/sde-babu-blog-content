```js
// ans to store the counter
    private int ans = 0;
    public int pathSum(TreeNode root, int sum) {
        if(root == null) return 0;
        // use long to handle the overflow and check for the path using helper function
        helper(root, (long)sum);
        // repeat for left and right
        pathSum(root.left, sum);
        pathSum(root.right, sum);
        
        return ans;
    }

    public void helper(TreeNode root, long sum) {
        if (root == null) return;
        if (sum == root.val) ans++;
        sum = sum-root.val;
        helper(root.left, sum);
        helper(root.right, sum);
    }
```
