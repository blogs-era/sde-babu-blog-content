public TreeNode construct(int[] preorder, int[] postorder,HashMap<Integer,Integer> postorder_map,int pre_start,int pre_end,int post_start,int post_end){
        if(pre_start > pre_end || post_start > post_end)return null;
        TreeNode root = new TreeNode(preorder[pre_start]);
        
        if(pre_start==pre_end)return root;// to cover the case of when only one element left and no next node present
        
        int next_node_index = postorder_map.get(preorder[pre_start+1]);
        int number_of_element_in_left = next_node_index-post_start+1;
        
        root.left = construct(preorder,postorder,postorder_map,pre_start+1,pre_start+number_of_element_in_left,post_start,next_node_index);
        root.right = construct(preorder,postorder,postorder_map,pre_start+number_of_element_in_left+1,pre_end,next_node_index+1,post_end-1);
        
        return root;
    }
    
    public TreeNode constructFromPrePost(int[] preorder, int[] postorder) {
        int n = postorder.length;
        HashMap<Integer,Integer> postorder_map = new HashMap<>();
        // put postorder <element,index> in postorder_map
        for(int i=0;i<n;i++)postorder_map.put(postorder[i],i);
        // construct tree
        return construct(preorder,postorder,postorder_map,0,n-1,0,n-1);
    }
