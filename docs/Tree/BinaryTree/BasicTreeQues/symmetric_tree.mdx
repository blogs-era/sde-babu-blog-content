/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isSymmetricUtil(TreeNode left,TreeNode right){
        if(left==null || right==null){
            // if both are null then true
            return left==right;
        }
        // check for the value
        if(left.val!=right.val)return false;
        // now compare left.left with right.right
        // and comapre left.right and right.left
        return isSymmetricUtil(left.left,right.right) && isSymmetricUtil(left.right,right.left);
    }
    public boolean isSymmetric(TreeNode root) {
        return isSymmetricUtil(root.left,root.right);
    }
}
