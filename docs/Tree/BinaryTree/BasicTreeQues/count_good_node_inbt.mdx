// recursive function
    public int dfs(TreeNode root,int maxValue){
        //base case
        if(root==null)return 0;
        // preorder
        int result = maxValue<=root.val ? 1: 0;
        maxValue = Math.max(root.val,maxValue);
        return result+dfs(root.left,maxValue)+dfs(root.right,maxValue);
    }
    public int goodNodes(TreeNode root) {
        // initially the max value is root.val
        return dfs(root,root.val);
    }
