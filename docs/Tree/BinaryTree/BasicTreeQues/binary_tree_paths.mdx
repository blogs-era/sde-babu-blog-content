### Statement

> Given the root of a binary tree, return all root-to-leaf paths in any order.

> A leaf is a node with no children.

### You can try it at

- https://leetcode.com/problems/binary-tree-paths/
 
### Example 1

![eg1](https://assets.leetcode.com/uploads/2021/03/12/paths-tree.jpg)
```
Input: root = [1,2,3,null,5]
Output: ["1->2->5","1->3"]
```

class Solution {
    public void dfs(TreeNode root,String path,List<String> paths){
        if(root==null)return;
        // leaf node means finish with one of the path
        if(root.left==null && root.right==null)paths.add(path+root.val);
        dfs(root.left,path+root.val+"->",paths);
        dfs(root.right,path+root.val+"->",paths);
    }
    public List<String> binaryTreePaths(TreeNode root) {
        List<String> paths = new ArrayList<>();
        if(root!=null)dfs(root,"",paths);
        return paths;
    }
}
