class Solution {
    public void dfs(TreeNode root,String path,List<String> paths){
        if(root==null)return;
        // leaf node means finish with one of the path
        if(root.left==null && root.right==null)paths.add(path+root.val);
        dfs(root.left,path+root.val+"->",paths);
        dfs(root.right,path+root.val+"->",paths);
    }
    public List<String> binaryTreePaths(TreeNode root) {
        List<String> paths = new ArrayList<>();
        if(root!=null)dfs(root,"",paths);
        return paths;
    }
}
