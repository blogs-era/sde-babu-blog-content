class Solution {
    public void dfs(List<List<Integer>> result,int[] candidates,int target,ArrayList<Integer> collection,int index){
        if(target < 0)return;   
        if(target==0){
            List<Integer> list = new ArrayList<>();
            for(Integer c: collection)list.add(c);
            result.add(list);
            return;
        }
        for(int i=index;i<candidates.length;i++){
            collection.add(candidates[i]);
            dfs(result,candidates,target-candidates[i],collection,i);
            collection.remove(collection.size()-1);
        }
    }
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        List<List<Integer>> result = new ArrayList<>();
        dfs(result,candidates,target,new ArrayList<>(),0);
        return result;
    }
}
