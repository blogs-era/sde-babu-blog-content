class Solution {
    public int threeSumClosest(int[] nums, int target) {
        int n = nums.length;
        int result = nums[0]+nums[1]+nums[n-1];// to contain unique triplets
        int k = target;
        Arrays.sort(nums);
        for(int i=0;i<n-2;i++){
            int low=i+1;int high = n-1;
            while(low < high){
                int sum = nums[i] + nums[low] + nums[high] ;
                if(sum < k)low++;
                else high--;

                // compare with current result
                if(Math.abs(sum-target) < Math.abs(result-target)){
                    result = sum;
                }

            }
        }
        return result;
    }
}
