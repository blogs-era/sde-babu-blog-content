class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        int n = nums.length;
        Set<List<Integer>> result = new HashSet<>();// to contain unique triplets
        int k = 0;// triplet sum should be equal to 0
        Arrays.sort(nums);
        for(int i=0;i<n-2;i++){
            int low=i+1;int high = n-1;
            while(low < high){
                int sum = nums[i] + nums[low] + nums[high] ;
                if(sum==k) result.add(Arrays.asList(nums[i], nums[low++], nums[high--]));
                else if(sum < k)low++;
                else high--;
            }
        }
        return new ArrayList<>(result);
    }
}
