class Solution {
    public void swap(int[] nums,int i,int j){
        int t = nums[i];
        nums[i] = nums[j];
        nums[j] = t;         
    }
    public void sortColors(int[] nums) {
       int n = nums.length;
       int l = 0,mid = 0,r = n-1;
       while(mid <= r){
           if(nums[mid]==0){
               // increase the left counter and reduce the boundary from left by increasing mid
               swap(nums,mid,l);
               mid++;
               l++;
           }else if(nums[mid]==1){
               // just increase the mid
               mid++;
           }else{
               // just decrease the boundary from right  
               swap(nums,mid,r);
               r--;
           }
       } 
    }
}
