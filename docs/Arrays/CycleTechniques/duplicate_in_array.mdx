```js
public int findDuplicate(int[] nums) {
    int slow = nums[0];
    int fast = slow;
    
    do{
        slow = nums[slow];
        fast = nums[nums[fast]];
    }while(fast!=slow);
    
    fast = nums[0];
    
    while(slow!=fast){
        slow = nums[slow];
        fast = nums[fast];
    }
    return slow;
}
```
