class Solution {
    /*
    arr: [1,1,1] and k = 2
    count_map<prefix_sum,freq>
    track all the prefix_sum
    and traverse the array and find whether prefix_sum[i]-k present in map if yes count+=count_map[prefix_sum-k]
    and add count_map[prefix_sum] in the map    
    */
    public int subarraySum(int[] nums, int k) {
        HashMap<Integer,Integer> count_map = new HashMap<>();
        count_map.put(0,1);
        int prefix_sum = 0,count = 0;
        for(int i=0;i<nums.length;i++){
            prefix_sum+=nums[i];// update the prefix sum
            if(count_map.containsKey(prefix_sum-k)){
                // that means found one subarray
                count+=count_map.get(prefix_sum-k);
            }
            // add the prefix sum in the count map
            count_map.put(prefix_sum,count_map.getOrDefault(prefix_sum,0)+1);
        } 
        return count;
    }
}
