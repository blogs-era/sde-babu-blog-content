class Solution {
    public int pivotIndex(int[] nums) {
        int n = nums.length;
        int[] prefix_sum = new int[n];
        for(int i=0;i<n;i++){
            if(i==0)prefix_sum[i] = nums[i];
            else prefix_sum[i] = prefix_sum[i-1]+nums[i];
        }       
        int ans = -1;
        int right_sum = 0;
        for(int i=n-1;i>=0;i--){
            int left_sum = i==0 ? 0 : prefix_sum[i-1]; 
            if(left_sum==right_sum)ans = i;
            right_sum+=nums[i];
        }
        return ans;
    }
}
