class Solution {
    public List<Integer> spiralOrder(int[][] matrix) {
        int n = matrix.length,m = matrix[0].length; 
        int t = 0,b = n-1,l = 0,r = m-1;
        List<Integer> result = new ArrayList<>();
        while(t <= b && l <= r){
            for(int i=l;i<=r;i++)result.add(matrix[t][i]);
            t++;
            for(int i=t;i<=b;i++)result.add(matrix[i][r]);
            r--;
            // now case to be noted if we left with [1,2] and we adde both and l++ but t<=b so duplicacy is there
            if(t > b || l > r)break;
            for(int i=r;i>=l;i--)result.add(matrix[b][i]);
            b--;
            for(int i=b;i>=t;i--)result.add(matrix[i][l]);
            l++;
        } 
        return result;
    }
}
