class Solution {
    public int majorityElement(int[] nums) {
        int num = nums[0],c = 1;
        for(int i=1;i<nums.length;i++){
            if(nums[i]==num){
                c++;
            }else{
                if(c>0)c--;
                else {
                    num = nums[i];
                    c=1;
                }
            }
        }
        return num;
    }
}
