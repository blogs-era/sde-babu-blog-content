class Solution {
    // go to each water and get the nearest land and compare to maximum 
    // simple multisource bfs (optimal approach)
    public int maxDistance(int[][] grid) {
        int n = grid.length,m = grid[0].length;
        Queue<int[]> q = new LinkedList<>();
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j]==1)
                q.add(new int[]{i,j});
            }
        }
        int[] dx = {0,0,-1,1};
        int[] dy = {1,-1,0,0}; 
        int level = -1;
        while(!q.isEmpty()){
            int size = q.size();
            for(int k=0;k<size;k++){
                int[] point = q.poll();
                for(int i=0;i<4;i++){
                    int x = dx[i]+point[0];
                    int y = dy[i]+point[1];
                    if(x>=0 && y>=0 && y<m && x<n && grid[x][y]==0){
                        grid[x][y] = 1;
                        q.add(new int[]{x,y});
                    }
                }
            }
            level++;
        }   
        return level==0 ? -1 :level;
    }
}
