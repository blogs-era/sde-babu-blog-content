### Recursive Code

```js
class Solution {
    public int dfs(int[] coins, int amount,int _count){
        if(amount < 0)return Integer.MAX_VALUE;
        if(amount==0)return _count;
        int min_number_of_coins = Integer.MAX_VALUE;
        for(int coin: coins){
            int c = dfs(coins,amount-coin,_count+1);
            min_number_of_coins = Math.min(c,min_number_of_coins);
        }
        return min_number_of_coins;
    }
    public int coinChange(int[] coins, int amount) {
        int count = dfs(coins,amount,0);
        return count==Integer.MAX_VALUE ? -1: count;
    }
}
```

### Iterative Method

```js
class Solution {
    public int coinChange(int[] coins, int amount) {
        int[] dp = new int[amount+1];// dp for amount from 0 - amount
        Arrays.fill(dp,Integer.MAX_VALUE);// it should be max to get the min count
        dp[0] = 0;// base case
        for(int i=0;i<amount;i++){
            if(dp[i]==Integer.MAX_VALUE)continue;
            for(int coin: coins){
                if(i+coin>=0 && i+coin <= amount){
                    dp[i+coin] = Math.min(dp[i+coin],dp[i]+1);
                }
            }
        }
        return dp[amount]==Integer.MAX_VALUE ? -1: dp[amount];
    }
}
```
