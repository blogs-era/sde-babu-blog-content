class Solution {
    public int longestValidParentheses(String s) {
        if(s.length()==0)return 0;
        int maxLen = 0;
        Stack<Integer> st = new Stack<>();
        st.push(-1);
        // loop through the string
        for(int i=0;i<s.length();i++){
            if(s.charAt(i)=='('){
                st.push(i);  
            }else{
                int top = st.peek();
                if(top!=-1 && s.charAt(top)=='('){
                    st.pop();
                    maxLen = Math.max(maxLen,i-st.peek());
                }else{
                    st.push(i);
                }
            }
        }
        return maxLen;
    }
}
