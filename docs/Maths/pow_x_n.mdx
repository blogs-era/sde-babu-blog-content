class Solution {
    public double dfs(double x,int n){
        if(x==0)return 0;
        if(n==0)return 1;
        double res = dfs(x,n/2);
        res = res*res;
        return n%2==0 ? res : x*res;
    }
    public double myPow(double x, int n) {
        double result = dfs(x,Math.abs(n));
        return n<0 ? 1/result : result;
    }
}
