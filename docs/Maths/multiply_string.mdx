class Solution {
    public String reverse(String str){
        String reversed = "";
        for(Character c: str.toCharArray()){
            reversed = c+reversed;
        }
        return reversed;
    }
    public String multiply(String num1, String num2) {
        if(num1.equals("0") || num2.equals("0"))return "0";
        int[] res = new int[num1.length()+num2.length()];
        // reverse both the strings
        num1 = reverse(num1);
        num2 = reverse(num2);
        // multiply
        for(int i=0;i<num1.length();i++){
            for(int j=0;j<num2.length();j++){
                // int val = (num1.charAt(i)-'0')*(num2.charAt(j)-'0');
                // System.out.println(val);
                res[i+j] += (num1.charAt(i)-'0')*(num2.charAt(j)-'0');
                res[i+j+1] += res[i+j]/10;
                res[i+j] = res[i+j]%10;
            }   
        }
        String out = "";
        //remove trailing zeroes and create to the result
        int index = num1.length()+num2.length()-1;
        while(index >=0 && res[index]==0)index--;
        for(int i=0;i<=index;i++){
            out = res[i]+out;
        }
        return out;
    }
}
