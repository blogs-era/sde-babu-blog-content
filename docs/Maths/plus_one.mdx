class Solution {
    public int[] plusOne(int[] digits) {
        int n = digits.length;
        digits[n-1] = digits[n-1]+1;// add 1 to the end 

        int[] res = new int[digits.length+1];// with one more digit
        int carry = 0;
        for(int i=n-1;i>=0;i--){
            int val = digits[i]+carry;// value
            carry = val/10;// carry
            res[i+1] = val%10; // store in result
        }
        if(carry > 0)res[0] = carry;
        // take care of trailing zeros
        return res[0]==0 ? Arrays.copyOfRange(res, 1, res.length) : res;
    }
}
