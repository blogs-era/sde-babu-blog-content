class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int n = cardPoints.length;
        int win_size = n-k;
        int sum = 0;
        for(int i=0;i<win_size;i++){
            sum+=cardPoints[i];
        }
        int minsum = sum;   
        int start = 0;
        int i = win_size;
        while(i < n){
            System.out.println(sum);
            sum = sum-cardPoints[start++]+cardPoints[i++];
            minsum = Math.min(minsum,sum);
        }
        int total = 0;
        for(int c: cardPoints)total+=c;      
        return total - minsum;     
    }
}
