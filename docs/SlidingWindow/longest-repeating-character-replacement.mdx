class Solution {
    public int characterReplacement(String s, int k) {
        int left = 0;
        int[] freq = new int[256];
        int max_freq = 0;// keep track of the most frequent character at every point
        int res = 0;
        for(int r=0;r<s.length();r++){
            freq[s.charAt(r)]++;// update the frequency map
            max_freq = Math.max(freq[s.charAt(r)],max_freq);// track the max_freq
            // reduce the left pointer means reduce the window size until we met the condition of less frequent element to be less than equals to k
            while(left < r && (r-left+1) - max_freq > k){
              freq[s.charAt(left)]--;
              left++;
              // note: we are not updating the max_freq because that will not effect the max_length/result
            }
            res = Math.max(res,r-left+1);
        }
        return res;
    }
}
