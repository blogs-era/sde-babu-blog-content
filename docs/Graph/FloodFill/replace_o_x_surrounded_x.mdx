### Statement

> Given a matrix mat of size N x M where every element is either ‘O’ or ‘X’.
Replace all ‘O’ with ‘X’ that are surrounded by ‘X’.
A ‘O’ (or a set of ‘O’) is considered to be by surrounded by ‘X’ if there are ‘X’ at locations just below, just above, just left and just right of it.

```js
static void flood_fill(int x,int y,int n,int m,char mat[][]){
        if(x<0 || y<0 || x>=n || y>=m || mat[x][y]!='-')return;
        mat[x][y] = 'O';
        
        int[] dx = {-1,1,0,0};
        int[] dy = {0,0,-1,1};
        for(int i=0;i<4;i++){
            int new_x = x+dx[i],new_y =y+dy[i];
            flood_fill(new_x,new_y,n,m,mat);
        }
        return;
    }
    static char[][] fill(int n, int m, char mat[][])
    {
        // replace all the 'O' with '-'
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(mat[i][j]=='O')mat[i][j] = '-';
            }
        }
        
        // travrese through grid and recurse for each corner '-' and fill it with 'O'
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if((i==0 || j==0 || i==n-1 || j==m-1) && mat[i][j]=='-'){
                    flood_fill(i,j,n,m,mat);
                }
            }
        }
        // those '-' which are unchanged just make them 'X'
        // replace all the '-' with 'X'
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(mat[i][j]=='-')mat[i][j] = 'X';
            }
        }
        
        return mat;
    }
```
