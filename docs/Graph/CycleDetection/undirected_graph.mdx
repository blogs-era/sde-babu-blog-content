class Solution {
    public boolean isCycleUtil(ArrayList<ArrayList<Integer>> adj,int node,int par,boolean[] vis){
        // visit the node
        vis[node] = true;
        ArrayList<Integer> neighbours = adj.get(node);
        for(Integer neigh: neighbours){
            if(!vis[neigh]){
                if(isCycleUtil(adj,neigh,node,vis))return true;
            }else{
                // visited
                if(par!=neigh)return true;
            }
        }
        return false;
    }
    public boolean isCycle(int V, ArrayList<ArrayList<Integer>> adj) {
        boolean[] vis = new boolean[V];
        // loop through all the components
        for(int i=0;i<V;i++){
            if(!vis[i]){
                // if cycle then return true 
                if(isCycleUtil(adj,i,-1,vis))return true;
            }
        }
        return false;
    }
}
