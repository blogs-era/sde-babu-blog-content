class Solution {
    public int countSubstrings(String s) {
        int n = s.length();
        boolean[][] table = new boolean[n][n];
        // we use a table to check the number of palindromic substrings
        /*str = abc
            a   b   a

        a   1   0   1

        b   0   1    0

        a   0   0    1

        ouput = 4
        */
        int count = 0;
        for(int gap=0;gap<n;gap++){
            for(int i=0,j=gap;i<n-gap && j < n;i++,j++){
                if(gap==0)table[i][j] = true;// means only one char is always palindrome
                else if(gap==1){
                    table[i][j] = s.charAt(i)==s.charAt(j);// means if 2 char is same then ok
                }else{
                    table[i][j] = s.charAt(i)==s.charAt(j) && table[i+1][j-1];
                }
                // increase the count if true
                if(table[i][j])count++;
            }
        }
        return count;
    }
}
