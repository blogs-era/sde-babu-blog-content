class Solution {
    public String minRemoveToMakeValid(String s) {
        Stack<Integer> stack = new Stack<>();
        int index = 0;
        boolean[] valid = new boolean[s.length()];
        for(Character c: s.toCharArray()){
            // if c is not baracket then include this 
            if(c!='(' && c!=')'){
                valid[index] = true;
            }
            // if opening bracket then push
            else if(c=='('){
                stack.push(index);
            }
            // closing bracket
            else{
                // if stack is not empty then
                if(!stack.isEmpty()){
                    valid[stack.pop()] = true;
                    valid[index] = true;
                }
            }
            index++;
        }   
        // all the valid char add it to the result 
        String out = "";
        for(int i=0;i<s.length();i++){
            if(valid[i]){
                System.out.print(s.charAt(i)+" ");
                out+=s.charAt(i);
            }
        }
        return out;
    }
}
