```js
    public ListNode oddEvenList(ListNode head) {
        // if only one and 2 node then no need
        if(head==null || head.next==null)return head;
        // odd_start and end will be the first node and even_start and end will be the second node
        ListNode os = head,es=head.next;
        ListNode oe = head,ee = head.next;
        // ex> 1 -> 2 -> 3 -> 4 -> 5 -> null 
        
        while(oe!=null && oe.next!=null && ee!=null && ee.next!=null){
            // connect odd index element to odd_list
            oe.next = oe.next.next;
            oe = oe.next;
            // connect even index element to odd_list
            ee.next = ee.next.next;
            ee = ee.next;
        }
        // after the while loop
        // 1 -> 3 -> 5 -> null and 2 -> 4 -> null
            
        oe.next = es;
        return os;
    }

```
