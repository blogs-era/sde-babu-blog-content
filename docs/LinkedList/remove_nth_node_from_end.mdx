```js
 public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode a = dummy;//previos node used for deletion
        ListNode b = dummy;
        
        // now be is at n steps from origin
        for(int i=0;i<n;i++){
            b = b.next;
        }
        // now move b and a until b is null 
        // this is because we have to move count-n steps from origin
        while(b.next!=null){
            a = a.next;
            b = b.next;
        }
        a.next = a.next.next;// delete the node
        // System.out.println(a.val);
        return dummy.next;
    }
```
