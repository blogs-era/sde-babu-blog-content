```js
 public ListNode reverseBetween(ListNode head, int left, int right) {
        // part 1 we mark the leftPrev node , dummy node and curr node and prev node to null
        ListNode dummy = new ListNode(0);
        ListNode prev = null;
        dummy.next = head;
        ListNode curr = head;
        ListNode leftPrev = dummy;
        for(int i=0;i<left-1;i++){
            curr = curr.next;
            leftPrev = leftPrev.next;
        }
        // part 2 we need to reverse the section
        for(int i=0;i<right-left+1;i++){
            ListNode next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        // after part2 we can just link the linkedlist
        leftPrev.next.next = curr;
        leftPrev.next = prev;
        
        return dummy.next;
    }

```
