class Solution {
    public int findMin(int[] nums) {
        int left = 0,right = nums.length-1;
        while(left<=right){
            int mid = (left+right)/2;
            // first check if nums[mid]==nums[right]
            if(nums[mid]==nums[right]){
                right--;// then remove the duplicate from last
            }
            // check if right sorted portion means nums[mid] is less than nums[right]
            else if(nums[mid] < nums[right]){
                // now element from mid+1 to right are useless
                right = mid;
            }
            // left sorted portion means nums[mid] is bigger than nums[right] 
            else{
                // now element from left to mid are useless 
                left = mid+1;
            }
        } 
        return nums[left];
    }
}
