class Solution {
    public int findPeakElement(int[] nums) {
        // just we have to remember 2 cases for intution behind it
        // that if array is increasing from left to right then rightmost is the ans 
        // if decreasing then leftmost is the ans
        // hence if next to mid is greater then we assume to be increasing and move rightwards
        int l = 0, r = nums.length-1;
        // l<=r -> can give tle: try this with eg: [6,4] where l = 0, r = 1, m = 0
        // 6 > 4 means decreasing means 6 can be the ans so r = 0 now
        // now l = 0 r = 0 m = 0 again it compare with left element and again m set to 0(loop)
        while(l < r){
            int m = (l+r)/2;
            if(nums[m] < nums[m+1]){
                // increasing case means our ans can lie b/w mid+1 to rightmost
                l = m+1;
            }else{
                r = m;
            }
        }
        return l;
    }
}
