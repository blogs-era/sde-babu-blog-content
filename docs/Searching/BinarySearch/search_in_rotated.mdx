```js
    public int search(int[] nums, int target) {
        int l = 0,r = nums.length-1;// init left and right 
        int mid = (l+r)/2;// init the mid
        while(l<=r){
            if(nums[mid]==target)return mid;// we found the element
            // first check whether the left sorted portion
            if(nums[l]<=nums[mid]){
                // can we move right
                if(target > nums[mid] || target < nums[l]){
                    l = mid+1;
                }
                // we have to move to left
                else{
                    r = mid-1;
                }
            }
            // right sorted portion
            else{
                // can we move to left
                if(target > nums[r] || target < nums[mid]){
                    r = mid-1;
                }
                // we have to move to right
                else{
                    l = mid+1;
                }
            }
            mid = (l+r)/2;// update the mid
            
        }
        return -1;// means no result found in arr
    }

```
