    public int sumArr(int[] arr){
        int sum = 0;
        for(int a: arr)sum+=a;
        return sum;
    }
    public int canCompleteCircuit(int[] gas, int[] cost) {
        int n = gas.length;
        //gas = [1,2,3,4,5], cost = [3,4,5,1,2]
        //our sum of gas should be bigger than cost otherwise we never complete the loop
        if(sumArr(cost) > sumArr(gas))return -1;
        
        // track the remaining_gas and starting_point
        int start = 0;
        int remaining_gas = 0;
        
        for(int i=0;i<n;i++){
            remaining_gas+=(gas[i]-cost[i]);// add current gas - cost
            if(remaining_gas < 0){
                remaining_gas = 0;// again start
                start = i+1;// start is the next pointer
            }
        }
        
        return start;
    }
