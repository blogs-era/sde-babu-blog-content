class Solution {
    /*
    case to remember: [[1,100],[1,11],[2,12],[11,22]]
    now in normal case we set the range to be 1  -  100
    but in that case our output is 3 because all are in range 1 - 100
    hence when we found a case like this [1,100] [1,11] then don't blindly remove the [1,11]
    compare the end pointer of both and set the end to be min of both ends
    */
    public int eraseOverlapIntervals(int[][] intervals) {
        int count = 0;
        Arrays.sort(intervals,(a,b) -> a[0]-b[0]);
        int end = intervals[0][1];
        for(int i=1;i<intervals.length;i++){
            if(intervals[i][0]>=end){
                end = intervals[i][1];
            }else{
                end = Math.min(end,intervals[i][1]);
                count++;
            }
        }
        return count;
    }
}
