public static int maxOnes (int Mat[][], int N, int M)
{
    int col = M-1;
    int ans = -1;
    for(int row=0;row<N;row++){
        while(col >= 0 && Mat[row][col]==1){
            col--;
            ans = row;
        }
    }
    return ans;
}
