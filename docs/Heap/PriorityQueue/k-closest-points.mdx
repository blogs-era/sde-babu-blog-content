### Example 1

```js
class Solution {
    // O(NLOG(K))
    public List<Integer> findClosestElements(int[] arr, int k, int x) {
        List<Integer> result = new ArrayList<>();// result
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> b[1]-a[1]);// int[] -> {original_elem,difference}
        // loop through array and get k smallest elem
        for(int i=0;i<arr.length;i++){
            int diff = Math.abs(arr[i]-x);// difference so that we get distance from origin
            if(pq.size() < k){
                pq.add(new int[]{arr[i],diff});
            }else{
                int[] node = pq.peek();
                if(node[1] > diff){
                    pq.poll();
                    pq.add(new int[]{arr[i],diff});
                }
            }
        }
        // extract elem from the pq and store in arraylist
        while(!pq.isEmpty()){
            result.add(pq.poll()[0]);
        }
        Collections.sort(result);// result should be in sorted fashion
        return result;
    }
}
```
