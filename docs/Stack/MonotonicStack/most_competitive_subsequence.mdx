class Solution {
    public int[] mostCompetitive(int[] nums, int k) {
        Stack<Integer> stack = new Stack<>();
        int operations_left = nums.length - k;
        for(int i=0;i<nums.length;i++){
            while(!stack.isEmpty() && nums[i] < stack.peek() && operations_left > 0){
                stack.pop();
                operations_left--;
            }
            stack.add(nums[i]);
        }
        int[] result = new int[k]; 
        while(stack.size() > k)stack.pop();
        int index = k-1;
        while(!stack.isEmpty())result[index--] = stack.pop();
        return result;
    }
}
