class Solution {
    public int evaluate(int part1,int part2,String operator){
        char c = operator.charAt(0);
        switch(c){
            case '+':
                return part1+part2;
            case '-':
                return part1-part2;
            case '*':
                return part1*part2;
            case '/':
                return part1/part2;
        }
        return -1;        
    }
    public int evalRPN(String[] tokens) {
        Stack<Integer> s = new Stack<>();
        for(String token: tokens){
            // if operator then evaluate
            if(token.equals("*") || token.equals("+") || token.equals("/") || token.equals("-")){
                int part2 = s.pop(),part1 = s.pop();// take both the operand
                int evaluate_res = evaluate(part1,part2,token);
                s.push(evaluate_res);
                continue;
            }
            // else if number then push to the stack
            s.push(Integer.parseInt(token));
        }  
        return s.pop();
    }
}
